\begin{align*} \lim_{k \rightarrow \infty} \frac{\left| a_{k+1} \right|} {\left| a_k \right|} \amp = \lim_{k \rightarrow \infty} \frac{\left| \frac{(x-2)^{k+1}}{(k+2)3^{k+1}} \right|} {\left| \frac{(x-2)^k}{(k+1)3^k} \right|}\\ \amp = \lim_{k \rightarrow \infty} \left| \frac{(x-2)^{k+1}}{(k+2)3^{k+1}} \cdot \frac{(k+1)3^k}{(x-2)^k} \right| \\ \amp = \lim_{k \rightarrow \infty} \left| \frac{(x-2)(k+1)}{3(k+2)} \right|\\ \amp = \left| \frac{x-2}{3} \right| \end{align*}
