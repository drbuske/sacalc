\begin{equation*} g'(-6) = \lim_{h\to 0} \frac{g(-6+h)-g(-6)}{h}, \end{equation*}
so
\begin{equation*} g'(-6) = \lim_{h\to 0}\frac{(4(-6+h)^2 + 6(-6+h)) - (4(-6)^2 + 6(-6))}{h}. \end{equation*}
Expanding the quadratic,
\begin{equation*} g'(-6) = \lim_{h\to 0} \frac{(144 - 48 h + 4 h^2 - 36 + 6 h) - (144 - 36)} {h} \end{equation*}
\begin{equation*} = \lim_{h\to 0} \frac{ -48 h + 4 h^2 + 6 h }{h} = \lim_{h\to 0} -48 + 6 + 4 h \end{equation*}
\begin{equation*} = -48 + 6 = -42. \end{equation*}
