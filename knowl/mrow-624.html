\begin{align*} \int \frac{\sqrt{4-x^2}}{x} \ dx = \mathstrut \amp \int \frac{\sqrt{4-(2\sin\theta)^2}}{2\sin\theta} \ \cdot 2\cos\theta \ d\theta\\ = \mathstrut \amp \int \frac{2\cos^2\theta}{\sin\theta} \ d\theta\\ = \mathstrut \amp \int \frac{2(1-\sin^2\theta)}{\sin\theta} \ d\theta\\ = \mathstrut \amp \int 2\csc\theta - 2\sin\theta \ d\theta\\ = \mathstrut \amp 2\ln|\csc\theta - \cot\theta| + 2\cos\theta + C\\ = \mathstrut \amp 2\ln\left\lvert\frac{2}{x} - \frac{\sqrt{4-x^2}}{x}\right\rvert + \sqrt{4-x^2} + C. \end{align*}
