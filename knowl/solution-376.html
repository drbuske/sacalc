\begin{equation*} \int_{1}^{3}\,\sqrt{1 + {81\over 4} x}\,dx = {{4\over {81\over 4}}} \int_{21.25}^{61.75}\,\sqrt{w}\,dw = {{8\over 0}} ( 61.75^{3/2} - 21.25^{3/2} ). \end{equation*}
