\begin{equation*} f''(x) = -4\sin\!\left(2x\right),\qquad \hbox{so}\quad f''(0) = 0, \end{equation*}

\begin{equation*} f^{4}(x) = 16\sin\!\left(2x\right),\qquad \hbox{so}\quad f^{4}(0) = 0, \end{equation*}
\begin{equation*} f^{5}(x) = 32\cos\!\left(2x\right),\qquad \hbox{so}\quad f^{5}(0) = 32, \end{equation*}
